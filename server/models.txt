The required entities:

1. Customer
    * userId       (String)
    * username     (String)
    * password     (String)
    * firstName    (String)
    * lastName     (String)
    * email        (String)
    * role         (Number)

2. Restaurant
    * restaurantId              (String)
    * Name                      (String)
    * businessEmail             (String Email)
    * password                  (String)
    * url                       (String Email)
    * address {                 (String Object)
         * city                 (String)
         * country              (String)
         * province             (String)
         * street               (String)
         * postalCode           (String)
    }                   
    * contactInfo               (Number)
    * about                     (String)
    * rating                    (Number)
    * menu                      (String)
    * logo                      (String Img)
    * bannerImg                 (String Img)
    * businessHours[] {         (JSON Array)
        * day: [                (ArrayOArray)  // let's go with 24 hours format to avoid complexity in UI
            *starTime,          (String)
            *endTime            (String)
         ]         
    }
    * seatingArrangements[] {   (JSON Array)
        *tableNumber            (Number)
        *tableCapacity          (Number)
    }

3. Menu
    * menuId           (String)
    * Restaurant       --ref (Obj Ref)
    * menuName         (String)
    * category         (String)
    * menuItems[] {    (JSON Array)
        *itemName      (String)
        *itemImage     (String Img)
        *about         (String)
        *price         (Number)
    }

4. Reservation
    * reservationId    (String)
    * Restaurant       --ref (Obj Ref)
    * User             --ref (Obj Ref)
    * date             (date)
    * slotInterval     (datetime)
    * paymentInfo      (String)
    * tableNumber      (Number)
    * status           (String)
